def merge_sort(orders):
    if len(orders) <= 1:
        return orders
    # Find the middle point of the list to divide it into two halves.
    mid = len(orders) // 2  # // integer division operator
    left_half = orders[:mid]
    right_half = orders[mid:]
    left_sorted = merge_sort(left_half)
    right_sorted = merge_sort(right_half)
    return merge(left_sorted, right_sorted)

def merge(left_half, right_half):
    merged_list = []
    i = 0  # Pointer for the left_half
    j = 0  # Pointer for the right_half
    # Compare elements from both lists and add the smaller one to the merged list.
    while i < len(left_half) and j < len(right_half):
        if left_half[i]['delivery_time'] < right_half[j]['delivery_time']:
            merged_list.append(left_half[i])
            i += 1  # Move the pointer for the left half
        else:
            merged_list.append(right_half[j])
            j += 1  # Move the pointer for the right half
    while i < len(left_half):
        merged_list.append(left_half[i])
        i += 1
    while j < len(right_half):
        merged_list.append(right_half[j])
        j += 1
    return merged_list

    online_orders = [
        {'order_id': 'A1', 'delivery_time': 45},

    ]
    print("Original list of orders:")
    for order in online_orders:
        print(f"Order {order['order_id']} -> Delivery Time: {order['delivery_time']} minutes")

    print("\nSorted list of orders by delivery time:")
    for order in sorted_orders:
